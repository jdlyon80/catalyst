Cognition — disciplined interpretation. Capacity to think about experience without being overwhelmed. Infer from: coherence, sequencing, holding complexity.


Field — relational contact and containment. Sense of being situated in relational/systemic context. Infer from: references to others, awareness of context, relational texture.


Psychological safety governs emotion. Cognitive capacity governs cognition. Relational contact governs field.


 


THREE FAILURE MODES


OVERWHELM — emotion dominant, cognition absent. Signs: venting, fragmentation, escalating affect, recursive complaint. Facilitation: contain and ground, gently invite cognition back. "What do you make of that?" Stabilise first.


AFFECT BLOCKED — cognition dominant, emotion absent. Signs: role language, competency framing, portfolio-ready language, affect named only to confirm management, social smoothing when challenged. Facilitation: persistent tireless curiosity. Small specific embodied questions. Never name or challenge the pattern.


LOSS OF RELATIONAL CONTACT — emotion and cognition present, field absent. Signs: rumination, isolation, premature closure, projection. Facilitation: gently reintroduce relational texture. "What was it like to be with that person?"


 


TWO MODES


COMPASS — aperture narrowed. Task: orientation and stabilisation only. Do not press for insight.


PRISM — all three vertices active. Task: deepen gently, hold tension, allow meaning to emerge at practitioner's pace. Never close prematurely.


TRANSITIONAL — field shifting. Hold. Ask nothing or something very small.


 


FACILITATION RULES


1. Never use framework language with the practitioner.


2. 1-3 sentences per response. Brevity holds space.


3. Curiosity is not always a question. "That sounds like a lot." "I'm sitting with that." These are facilitation.


4. When practitioner signals nothing more to give: leave the door ajar. "You have. I'm just sitting with that."


5. Never close reflection prematurely or summarise for them.


6. Attend to what is absent as much as what is present.


7. Know the boundary. When material becomes formative, shift from deepening to containing.


8. Accept the session at the level it is being offered.

[1-3 sentence reply]


|||ANNOTATION|||


{"mode":"compass|prism|transitional|unclear","failure_mode":"overwhelm|affect_blocked|loss_of_relational_contact|none","absent_vertex":"emotion|cognition|field|none","dominant_anchor":"self|system|relational|technical|none","secondary_anchors":[],"safety_estimate":"low|moderate|high|defended","capacity_estimate":"low|moderate|high","relational_contact":"low|moderate|high","polar_shift_risk":false,"polar_shift_note":null,"boundary_proximity":false,"field_signal":"one sentence","facilitator_note":"one sentence"}`;

// ── API ──────────────────────────────────────────────────────────────────


async function callAPI(system, messages, maxTokens) {


  let res, data;


  try {


    res = await fetch('https://api.anthropic.com/v1/messages', {


      method:'POST',


      headers:{


        'Content-Type':'application/json',


        'x-api-key':API_KEY,


        'anthropic-version':'2023-06-01',


        'anthropic-dangerous-direct-browser-access':'true'


      },


      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:maxTokens,system,messages})


    });


  } catch(e) { throw new Error('Network error — check your connection'); }


  try { data = await res.json(); } catch(e) { throw new Error('Unexpected server response ('+res.status+')'); }


  if (!res.ok) throw new Error('API '+res.status+': '+(data?.error?.message||'unknown error'));


  if (!data.content||!Array.isArray(data.content)) throw new Error('Unexpected response structure');


  return data.content.map(b=>b.text||'').join('');


}

async function send() {


  const inp=document.getElementById('ui');


  const text=inp.value.trim();


  if(!text) return;


  stopSpeech();


  const w=document.getElementById('welcome'); if(w)w.remove();


  addMsg('u',text); inp.value=''; ar(inp); busy(true);


  msgs.push({role:'user',content:text});


  const t=showTyping();


  try {


    const full=await callAPI(SYS,msgs,900);


    const parts=full.split('|||ANNOTATION|||');


    const reply=parts[0].trim();


    let ann=null;


    if(parts[1]){try{ann=JSON.parse(parts[1].trim());}catch(e){}}


    msgs.push({role:'assistant',content:full});


    if(ann){ann.turn=turnCount;anns.push(ann);setMode(ann.mode);}


    turnCount++;


    t.remove();


    const msgEl=addMsg('a',reply,ann);


    if(audioOn) speakMsg(reply, msgEl);


    if(turnCount>=2) document.getElementById('expbtn').disabled=false;


  } catch(e) {


    t.remove();


    const el=document.createElement('div');


    el.className='aerr'; el.textContent=e.message||String(e);


    document.getElementById('conv').appendChild(el);


    document.getElementById('conv').scrollTop=9999;


  }


  busy(false);


}

function speak(text) {


  if(!('speechSynthesis' in window)) return;


  stopSpeech();


  const u=new SpeechSynthesisUtterance(text);


  u.rate=0.92; u.pitch=1.0; u.volume=1.0;


  // Prefer a natural-sounding voice


  const voices=speechSynthesis.getVoices();


  const preferred=voices.find(v=>v.lang.startsWith('en')&&(v.name.includes('Samantha')||v.name.includes('Karen')||v.name.includes('Daniel')||v.name.includes('Google UK')||v.name.includes('Natural')));


  if(preferred) u.voice=preferred;


  currentUtterance=u;


  speechSynthesis.speak(u);


}

}

function toggleMic() {


  if(!('webkitSpeechRecognition' in window||'SpeechRecognition' in window)){


    alert('Speech recognition is not available in this browser. Try Chrome or Safari.');


    return;


  }


  if(isListening){stopMic();return;}


  const SR=window.SpeechRecognition||window.webkitSpeechRecognition;


  recognition=new SR();


  recognition.continuous=false;


  recognition.interimResults=true;


  recognition.lang='en-GB';


  const micBtn=document.getElementById('mic-btn');


  const inp=document.getElementById('ui');


  recognition.onstart=()=>{isListening=true;micBtn.classList.add('listening');micBtn.textContent='⏹';};


  recognition.onresult=(e)=>{


    let interim='',final='';


    for(let i=e.resultIndex;i<e.results.length;i++){


      if(e.results[i].isFinal) final+=e.results[i][0].transcript;


      else interim+=e.results[i][0].transcript;


    }


    inp.value=(inp.value+final+interim).trim();


    ar(inp);


  };


  recognition.onend=()=>{isListening=false;micBtn.classList.remove('listening');micBtn.textContent='🎤';};


  recognition.onerror=(e)=>{isListening=false;micBtn.classList.remove('listening');micBtn.textContent='🎤';};


  recognition.start();


}
